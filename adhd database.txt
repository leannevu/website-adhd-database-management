CREATE TABLE school (
schoolName varchar(50) NOT NULL,
schoolAddress varchar(50) NOT NULL,
PRIMARY KEY(schoolName)
);

CREATE TABLE student (
studentId int(11) NOT NULL,
studentName varchar(50) NOT NULL,
studentAddress varchar(50) NOT NULL,
schoolName varchar(50) NOT NULL,
PRIMARY KEY(studentId),
CONSTRAINT FK_student_schoolName FOREIGN KEY(schoolName) 
REFERENCES school(schoolName)
);

CREATE TABLE class (
className varchar(50) NOT NULL,
schoolName varchar(50),
teacherName varchar(50) NOT NULL,
CONSTRAINT PK_class PRIMARY KEY (className, teacherName),
CONSTRAINT FK_class_schoolName FOREIGN KEY(schoolName)
REFERENCES school(schoolName)
);

CREATE TABLE guardian (
studentId int(11) NOT NULL,
guardianName varchar(50) NOT NULL,
guardianNumber varchar(20),
guardianEmail varchar(30),
guardianRelation varchar(20),
CONSTRAINT PK_guardian PRIMARY KEY (studentId, guardianName),
CONSTRAINT FK_guardian_studentId FOREIGN KEY (studentId)
REFERENCES student(studentId)
);

CREATE TABLE assignment (
studentId int(11) NOT NULL,
assignmentName varchar(50) NOT NULL,
status varchar(20) NOT NULL,
due varchar(20) NOT NULL,
className varchar(50) NOT NULL,
CONSTRAINT PK_assignment PRIMARY KEY (studentId, assignmentName),
CONSTRAINT FK_assignment_studentId FOREIGN KEY (studentId)
REFERENCES student(studentId),
CONSTRAINT FK_assignment_className FOREIGN KEY (className)
REFERENCES class(className)
);
